#include <iostream>
using namespace std;

int main() {
    string num ;
    cin >> num ;
    bool a;
    for(int i : num){
        if(i == '7' || i == '4'){
            a = true;
        }
        else a = false;
    }
    if(a){
        cout << "YES";
    }
    else {
        cout << "NO";
    }
    
    return 0;
}
