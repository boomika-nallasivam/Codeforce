#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() {
    int s;
    cin >> s;

    int a1 = 0, a2 = 0;

    for(int i = 0; i < s; i++) {
        int a, b;
        cin >> a >> b;

        a1 -= a; 
        a1 += b;  
        a2 = max(a2, a1); 
    }

    cout << a2;

    return 0;
}
