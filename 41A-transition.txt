#include <bits/stdc++.h>
using namespace std;

int main() {
    string s;
    cin >> s;

    string s1 = s;  
    for(char &c : s) c = tolower(c);
    for(char &c : s1) c = tolower(c);

    reverse(s.begin(), s.end());     if(s == s1)
        cout << "YES";
    else
        cout << "NO";

    return 0;
}
