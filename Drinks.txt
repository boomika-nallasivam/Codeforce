#include <iostream>
#include <bits/stdc++.h>
using namespace std;


int main() {
    int s;
    cin >> s;
    
    double p;
    double sum =0;
    for(int i = 0 ; i<s ; i++){
        cin >> p;
        
        sum+=p;
    }
    double d = (double)sum / s;
    cout << d << endl;

    return 0;
}
