#include <iostream>
#include <bits/stdc++.h>
using namespace std;


int main() {
    int s;
    cin >> s;
    
    string p , cu;
    int c = 1;
    
    cin >> p;
    for(int i = 1 ; i<s ; i++){
       cin >> cu;
      
       if(cu!=p){
           c++;
       }
       p=cu;
    }
    cout << c << endl;

    return 0;
}
